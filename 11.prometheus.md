# 프로메테우스

```bash
# 네임스페이스 생성
kubectl create ns monitering

#프로메테우스가 쿠버네티스 API에 접근할 수 있는 권한 부여
vi prometheus-cluster-role.yaml

#적용
kubectl apply -f prometheus-cluster-role.yaml

#prometheus가 수집할 지표를 정의
vi prometheus-config-map.yaml

#적용
kubectl apply -f prometheus-config-map.yaml

#프로메테우스를 위한 파드 생성
vi prometheus-deployment.yaml

#적용
kubectl apply -f prometheus-deployment.yaml

#노드 내보내기 정의(정보수집역할)
vi prometheus-node-exporter.yaml

#적용
kubectl apply -f prometheus-node-exporter.yaml

#외부에서 접속할 수 있도록 노드포트 지정
vi prometheus-svc.yaml

#적용
kubectl apply -f prometheus-svc.yaml

#파드 확인
kubectl get pod -n monitoring
```

브라우저 확인 : http://마스터노드:30001
